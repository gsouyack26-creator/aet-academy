function glossModule(term){

  if(key.length<2)return -1;
  var esc=key.replace(/[.*+?^${}()|[\]\\]/g,'\\$&');
  var re;try{re=new RegExp('\\b'+esc+'\\b','i');}catch(e){re=null;}
  var ids=MODS.map(function(m){return m.id;}).sort(function(a,b){return a-b;});
  if(re){for(var i=0;i<ids.length;i++){if(re.test(_modText(ids[i])))return ids[i];}}
  for(var j=0;j<ids.length;j++){if(_modText(ids[j]).indexOf(key)>=0)return ids[j];}
  return -1;
}
