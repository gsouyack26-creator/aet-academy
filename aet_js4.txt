function simChecklist(){
  var stats=S.simStats||{};
  var done=SIMS.filter(function(s){return stats[s.id]||(S.sims||[]).indexOf(s.id)>=0;}).length;
  var cells=SIMS.map(function(s){
    var tried=stats[s.id]||(S.sims||[]).indexOf(s.id)>=0;
    var mark=tried?'&#10004;':'&#9675;';
    var col=tried?'var(--accent2)':'var(--txt2)';
    return '<div class="simchk" title="'+esc(s.name)+'" onclick="var e=$(\u0027sim-'+s.id+'\u0027);if(e&&e.scrollIntoView)e.scrollIntoView({behavior:\u0027smooth\u0027,block:\u0027start\u0027});" style="display:flex;align-items:center;gap:.35rem;padding:.3rem .5rem;border-radius:8px;cursor:pointer;background:var(--bg2);font-size:.8rem">'+
      '<span style="color:'+col+';font-weight:700">'+mark+'</span><span style="opacity:.85">'+s.icon+'</span><span style="white-space:nowrap;overflow:hidden;text-overflow:ellipsis">'+esc(s.name)+'</span></div>';
  }).join('');

function simActivityPanel(){

  tried.sort(function(a,b){return (stats[b.id].last||'').localeCompare(stats[a.id].last||'');});

/* ==== CodeForge ST interpreter (window.runSt) ported for live grading ==== */
/* runner_st.js — a small Structured Text (IEC 61131-3) interpreter.
   Executes a practical subset so ST challenges can be auto-graded:
   assignment (:=), IF/ELSIF/ELSE, CASE OF, FOR/WHILE/REPEAT, arrays,
   arithmetic (+ - * / MOD), comparison (= <> < <= > >=),
   boolean (AND OR NOT XOR), MIN/MAX/ABS and type-conversion functions.
   VAR/CONST blocks and comments are handled. Timers/function-blocks are
   NOT supported — those challenges stay guided. */
(function () {
  "use strict";

  var KW = { "IF":1,"THEN":1,"ELSIF":1,"ELSE":1,"END_IF":1,"CASE":1,"OF":1,
    "END_CASE":1,"FOR":1,"TO":1,"BY":1,"DO":1,"END_FOR":1,"WHILE":1,
    "END_WHILE":1,"REPEAT":1,"UNTIL":1,"END_REPEAT":1,"VAR":1,"VAR_INPUT":1,
    "VAR_OUTPUT":1,"VAR_TEMP":1,"VAR_GLOBAL":1,"END_VAR":1,"CONST":1,
    "END_CONST":1,"FUNCTION_BLOCK":1,"END_FUNCTION_BLOCK":1,"FUNCTION":1,"END_FUNCTION":1,"RETURN":1,
    "EXIT":1,"TYPE":1,"END_TYPE":1 };
  var OPWORD = { "AND":"&&","OR":"||","XOR":"xor","NOT":"!","MOD":"%","DIV":"/" };

  function stripComments(src) {
    var out = "", i = 0, n = src.length;
    while (i < n) {
      var two = src.substr(i, 2);
      if (two === "(*") { var e = src.indexOf("*)", i + 2); i = (e < 0) ? n : e + 2; continue; }
      if (two === "//") { var nl = src.indexOf("\n", i); i = (nl < 0) ? n : nl; continue; }
      if (src[i] === "'") {
        var j = i + 1; while (j < n && src[j] !== "'") j++;
        out += src.substring(i, j + 1); i = j + 1; continue;
      }
      out += src[i]; i++;
    }
    return out;
  }

  // ---- tokenizer ----
  function tokenize(src) {
    src = stripComments(src);
    var toks = [], i = 0, n = src.length;
    var isD = function (c) { return c >= "0" && c <= "9"; };
    var isA = function (c) { return (c >= "a" && c <= "z") || (c >= "A" && c <= "Z") || c === "_"; };
    while (i < n) {
      var c = src[i];
      if (c === " " || c === "\t" || c === "\r" || c === "\n") { i++; continue; }
      if (c === "'") { var j = i + 1; while (j < n && src[j] !== "'") j++; toks.push({ t: "str", v: src.substring(i + 1, j) }); i = j + 1; continue; }
      if (isD(c) || (c === "." && isD(src[i + 1]))) {
        var s = i; var hex = false;
        if (c === "1" && src[i + 1] === "6" && src[i + 2] === "#") { i += 3; var hs = i; while (i < n && /[0-9a-fA-F]/.test(src[i])) i++; toks.push({ t: "num", v: parseInt(src.substring(hs, i), 16) }); continue; }
        if (c === "2" && src[i + 1] === "#") { i += 2; var bs = i; while (i < n && /[01]/.test(src[i])) i++; toks.push({ t: "num", v: parseInt(src.substring(bs, i), 2) }); continue; }
        while (i < n && (isD(src[i]) || src[i] === "." || src[i] === "e" || src[i] === "E" || ((src[i] === "+" || src[i] === "-") && (src[i - 1] === "e" || src[i - 1] === "E")))) i++;
        toks.push({ t: "num", v: parseFloat(src.substring(s, i)) }); continue;
      }
      if (isA(c)) {
        var a = i; while (i < n && (isA(src[i]) || isD(src[i]))) i++;
        var w = src.substring(a, i), U = w.toUpperCase();
        // TIME literal like T#5s / T#500ms -> milliseconds number
        if ((U === "T" || U === "TIME") && src[i] === "#") { var ts = i + 1; i++; while (i < n && /[0-9smhdSMHD_.]/.test(src[i])) i++; toks.push({ t: "num", v: parseTime(src.substring(ts, i)) }); continue; }
        if (U === "TRUE") { toks.push({ t: "num", v: 1, b: true }); continue; }
        if (U === "FALSE") { toks.push({ t: "num", v: 0, b: true }); continue; }
        if (KW[U]) { toks.push({ t: "kw", v: U }); continue; }
        if (OPWORD[U]) { toks.push({ t: "op", v: OPWORD[U] }); continue; }
        toks.push({ t: "id", v: w }); continue;
      }
      var three = src.substr(i, 2);
      if (three === ":=" || three === "<=" || three === ">=" || three === "<>") { toks.push({ t: "op", v: three }); i += 2; continue; }
      if ("+-*/<>=(),;[].".indexOf(c) >= 0) { toks.push({ t: "op", v: c }); i++; continue; }
      i++; // skip anything unexpected (e.g. ':' in decls handled elsewhere)
      toks.push({ t: "op", v: c });
    }
    toks.push({ t: "eof" });
    return toks;
  }

  function parseTime(s) {
    var ms = 0, re = /([0-9.]+)\s*(ms|s|m|h|d)/gi, m;
    var any = false;
    while ((m = re.exec(s))) {
      any = true; var v = parseFloat(m[1]), u = m[2].toLowerCase();
      ms += u === "ms" ? v : u === "s" ? v * 1000 : u === "m" ? v * 60000 : u === "h" ? v * 3600000 : v * 86400000;
    }
    return any ? ms : parseFloat(s) || 0;
  }

  // ---- parser: produces statement list ----
  function parse(toks) {
    var p = 0;
    function peek() { return toks[p]; }
    function next() { return toks[p++]; }
    function isKw(v) { return toks[p].t === "kw" && toks[p].v === v; }
    function eatKw(v) { if (!isKw(v)) err("expected " + v + " got " + tokStr(toks[p])); p++; }
    function isOp(v) { return toks[p].t === "op" && toks[p].v === v; }
    function eatOp(v) { if (!isOp(v)) err("expected '" + v + "' got " + tokStr(toks[p])); p++; }
    function err(m) { throw new Error("Parse error: " + m); }

    function block(stops) {
      var out = [];
      while (peek().t !== "eof" && !(peek().t === "kw" && stops.indexOf(peek().v) >= 0)) {
        var s = statement();
        if (s) out.push(s);
      }
      return out;
    }

    function skipDecls() {
      // consume tokens up to matching END_VAR / END_CONST / END_TYPE
      var endTok = { "VAR": "END_VAR", "VAR_INPUT": "END_VAR", "VAR_OUTPUT": "END_VAR", "VAR_TEMP": "END_VAR", "VAR_GLOBAL": "END_VAR", "CONST": "END_CONST", "TYPE": "END_TYPE" };
      var open = next().v, want = endTok[open], decls = [];
      var FBT = { "TON":1,"TOF":1,"TP":1,"CTU":1,"CTD":1,"R_TRIG":1,"F_TRIG":1,"RS":1,"SR":1,"CTUD":1 };
      while (peek().t !== "eof" && !(peek().t === "kw" && peek().v === want)) {
        // capture `name : type [:= initExpr] ;`  (record FB type, user-FB type, or initializer)
        if (peek().t === "id") {
          var nm = next().v;
          if (isOp(":")) { next();
            var typeTok = peek(); var entry = { name: nm };
            if (typeTok.t === "id") { var TU = typeTok.v.toUpperCase(); if (FBT[TU]) entry.fbType = TU; else entry.typeName = TU; }
            while (peek().t !== "eof" && !isOp(";") && !isOp(":=") && !(peek().t === "kw" && peek().v === want)) next();
            if (isOp(":=")) { next(); entry.init = expr(); }
            decls.push(entry);
          }
        }
        while (peek().t !== "eof" && !isOp(";") && !(peek().t === "kw" && peek().v === want)) next();
        if (isOp(";")) next();
      }
      if (peek().t === "kw" && peek().v === want) next();
      return { type: "decls", kind: open, decls: decls };
    }

    function statement() {
      var tk = peek();
      if (tk.t === "kw") {
        switch (tk.v) {
          case "VAR": case "VAR_INPUT": case "VAR_OUTPUT": case "VAR_TEMP": case "VAR_GLOBAL": case "CONST": case "TYPE": return skipDecls();
          case "FUNCTION_BLOCK": return fbDefStmt();
          case "END_FUNCTION_BLOCK": next(); return null;
          case "FUNCTION": return fnDefStmt();
          case "END_FUNCTION": next(); return null;
          case "IF": return ifStmt();
          case "CASE": return caseStmt();
          case "FOR": return forStmt();
          case "WHILE": return whileStmt();
          case "REPEAT": return repeatStmt();
          case "RETURN": next(); if (isOp(";")) next(); return { type: "return" };
          case "EXIT": next(); if (isOp(";")) next(); return { type: "exit" };
        }
      }
      if (isOp(";")) { next(); return null; }
      // call-statement: Name( ... ) ;   (e.g. a timer/counter FB invocation)
      if (peek().t === "id" && toks[p + 1] && toks[p + 1].t === "op" && toks[p + 1].v === "(") {
        var callNode = primary(); if (isOp(";")) next(); return { type: "exprstmt", expr: callNode };
      }
      // assignment: lvalue := expr ;
      var lv = lvalue();
      eatOp(":="); var rhs = expr(); if (isOp(";")) next();
      return { type: "assign", target: lv, value: rhs };
    }

    function lvalue() {
      var id = next(); if (id.t !== "id") err("expected identifier");
      var node = { type: "var", name: id.v };
      while (isOp("[") || isOp(".")) {
        if (isOp("[")) { next(); var idx = expr(); eatOp("]"); node = { type: "index", base: node, index: idx }; }
        else { next(); var f = next(); node = { type: "field", base: node, name: f.v }; }
      }
      return node;
    }

    function ifStmt() {
      eatKw("IF"); var cond = expr(); eatKw("THEN");
      var then = block(["ELSIF", "ELSE", "END_IF"]);
      var clauses = [{ cond: cond, body: then }], els = null;
      while (isKw("ELSIF")) { next(); var c2 = expr(); eatKw("THEN"); clauses.push({ cond: c2, body: block(["ELSIF", "ELSE", "END_IF"]) }); }
      if (isKw("ELSE")) { next(); els = block(["END_IF"]); }
      eatKw("END_IF"); if (isOp(";")) next();
      return { type: "if", clauses: clauses, els: els };
    }

    function caseBody() {
      // a branch body runs until ELSE / END_CASE (keywords) or the next
      // numeric/string label (which starts the next branch).
      var out = [];
      while (peek().t !== "eof" && !isKw("ELSE") && !isKw("END_CASE") && peek().t !== "num" && peek().t !== "str") {
        var st = statement(); if (st) out.push(st);
      }
      return out;
    }
    function caseStmt() {
      eatKw("CASE"); var sel = expr(); eatKw("OF");
      var branches = [], els = null;
      while (!isKw("END_CASE") && !isKw("ELSE") && peek().t !== "eof") {
        var labels = [labelVal()];
        while (isOp(",")) { next(); labels.push(labelVal()); }
        eatOp(":");
        branches.push({ labels: labels, body: caseBody() });
      }
      if (isKw("ELSE")) { next(); els = block(["END_CASE"]); }
      eatKw("END_CASE"); if (isOp(";")) next();
      return { type: "case", sel: sel, branches: branches, els: els };
    }
    function labelVal() { var t = next(); if (t.t === "num") return t.v; if (t.t === "str") return t.v; err("bad case label"); }

    function fnDefStmt() {
      eatKw("FUNCTION");
      var nm = next(); var fnName = (nm.t === "id") ? nm.v : "FN";
      if (isOp(":")) { next(); if (peek().t === "id" || peek().t === "kw") next(); }
      var inputs = [], locals = [];
      while (peek().t === "kw" && (peek().v === "VAR_INPUT" || peek().v === "VAR" || peek().v === "VAR_TEMP")) {
        var d = skipDecls();
        if (d.kind === "VAR_INPUT") inputs = inputs.concat(d.decls);
        else locals = locals.concat(d.decls);
      }
      var body = block(["END_FUNCTION"]);
      if (isKw("END_FUNCTION")) next(); if (isOp(";")) next();
      return { type: "fndef", name: fnName, nameU: fnName.toUpperCase(), inputs: inputs, locals: locals, body: body };
    }
    function fbDefStmt() {
      eatKw("FUNCTION_BLOCK");
      var nm = next(); var fbName = (nm.t === "id") ? nm.v : "FB";
      var inputs = [], outputs = [], locals = [];
      while (peek().t === "kw" && (peek().v === "VAR_INPUT" || peek().v === "VAR_OUTPUT" || peek().v === "VAR" || peek().v === "VAR_TEMP")) {
        var d = skipDecls();
        if (d.kind === "VAR_INPUT") inputs = inputs.concat(d.decls);
        else if (d.kind === "VAR_OUTPUT") outputs = outputs.concat(d.decls);
        else locals = locals.concat(d.decls);
      }
      var body = block(["END_FUNCTION_BLOCK"]);
      if (isKw("END_FUNCTION_BLOCK")) next(); if (isOp(";")) next();
      return { type: "fbdef", name: fbName, nameU: fbName.toUpperCase(), inputs: inputs, outputs: outputs, locals: locals, body: body };
    }
    function forStmt() {
      eatKw("FOR"); var v = next().v; eatOp(":="); var from = expr(); eatKw("TO"); var to = expr();
      var by = null; if (isKw("BY")) { next(); by = expr(); }
      eatKw("DO"); var body = block(["END_FOR"]); eatKw("END_FOR"); if (isOp(";")) next();
      return { type: "for", v: v, from: from, to: to, by: by, body: body };
    }
    function whileStmt() { eatKw("WHILE"); var c = expr(); eatKw("DO"); var b = block(["END_WHILE"]); eatKw("END_WHILE"); if (isOp(";")) next(); return { type: "while", cond: c, body: b }; }
    function repeatStmt() { eatKw("REPEAT"); var b = block(["UNTIL"]); eatKw("UNTIL"); var c = expr(); eatKw("END_REPEAT"); if (isOp(";")) next(); return { type: "repeat", cond: c, body: b }; }

    // expression precedence: OR/XOR < AND < NOT < compare < add < mul < unary < primary
    function expr() { return orExpr(); }
    function orExpr() { var l = andExpr(); while (isOp("||") || isOp("xor")) { var o = next().v; var r = andExpr(); l = { type: "bin", op: o, l: l, r: r }; } return l; }
    function andExpr() { var l = cmpExpr(); while (isOp("&&")) { next(); var r = cmpExpr(); l = { type: "bin", op: "&&", l: l, r: r }; } return l; }
    function cmpExpr() { var l = addExpr(); while (isOp("=") || isOp("<>") || isOp("<") || isOp("<=") || isOp(">") || isOp(">=")) { var o = next().v; var r = addExpr(); l = { type: "bin", op: o, l: l, r: r }; } return l; }
    function addExpr() { var l = mulExpr(); while (isOp("+") || isOp("-")) { var o = next().v; var r = mulExpr(); l = { type: "bin", op: o, l: l, r: r }; } return l; }
    function mulExpr() { var l = unary(); while (isOp("*") || isOp("/") || isOp("%")) { var o = next().v; var r = unary(); l = { type: "bin", op: o, l: l, r: r }; } return l; }
    function unary() { if (isOp("!")) { next(); return { type: "not", e: unary() }; } if (isOp("-")) { next(); return { type: "neg", e: unary() }; } if (isOp("+")) { next(); return unary(); } return primary(); }
    function primary() {
      var tk = peek();
      if (isOp("(")) { next(); var e = expr(); eatOp(")"); return e; }
      if (tk.t === "num") { next(); return { type: "lit", v: tk.v }; }
      if (tk.t === "str") { next(); return { type: "lit", v: tk.v }; }
      if (tk.t === "id") {
        var name = next().v;
        if (isOp("(")) {
          next();
          // named args (FB style):  IN := expr, PT := expr
          if (peek().t === "id" && toks[p + 1] && toks[p + 1].t === "op" && toks[p + 1].v === ":=") {
            var named = {};
            while (!isOp(")") && peek().t !== "eof") {
              var an = next().v; eatOp(":="); named[an.toUpperCase()] = expr();
              if (isOp(",")) next();
            }
            eatOp(")");
            return { type: "call", name: name, nameU: name.toUpperCase(), named: named };
          }
          var args = []; if (!isOp(")")) { args.push(expr()); while (isOp(",")) { next(); args.push(expr()); } } eatOp(")");
          return { type: "call", name: name, nameU: name.toUpperCase(), args: args };
        }
        var node = { type: "var", name: name };
        while (isOp("[") || isOp(".")) { if (isOp("[")) { next(); var idx = expr(); eatOp("]"); node = { type: "index", base: node, index: idx }; } else { next(); var f = next(); node = { type: "field", base: node, name: f.v }; } }
        return node;
      }
      throw new Error("Parse error: unexpected " + tokStr(tk));
    }

    var prog = block([]);
    if (peek().t !== "eof") throw new Error("Parse error: trailing input " + tokStr(peek()));
    return prog;
  }
  function tokStr(t) { return t.t === "eof" ? "<end>" : (t.v != null ? String(t.v) : t.t); }

  // ---- evaluator ----
  function keyOf(node) {
    if (node.type === "var") return node.name;
    if (node.type === "field") return keyOf(node.base) + "." + node.name;
    return null;
  }
  function run(prog, env) {
    var STEPS = { n: 0, max: 2000000 };
    var FBDEFS = {}; for (var _fi = 0; _fi < prog.length; _fi++) if (prog[_fi].type === "fbdef") FBDEFS[prog[_fi].nameU] = prog[_fi];

    var fnSeq = 0;

    function tick() { if (++STEPS.n > STEPS.max) throw new Error("Runaway loop (>2M steps) — check loop bounds"); }
    function get(name) { return env[name]; }
    function evalNode(nd) {
      switch (nd.type) {
        case "lit": return nd.v;
        case "var": { var v = env[K(nd.name)]; return v === undefined ? 0 : v; }
        case "field": { var o = env[curPrefix + keyOf(nd)]; return o === undefined ? 0 : o; }
        case "index": { var arr = evalNode(nd.base); var i = evalNode(nd.index) | 0; if (arr === 0 || arr === undefined) return 0; if (!Array.isArray(arr)) throw new Error("not an array: " + keyOf(nd.base)); if (i < 0) throw new Error("array index out of range [" + i + "]"); return arr[i] === undefined ? 0 : arr[i]; }
        case "neg": return -evalNode(nd.e);
        case "not": return truthy(evalNode(nd.e)) ? 0 : 1;
        case "call": { var nu = nd.nameU || nd.name; if (FNDEFS[nu]) return runUserFn(FNDEFS[nu], (nd.args || []).map(evalNode)); return callFn(nu, (nd.args || []).map(evalNode)); }
        case "bin": {
          var l = evalNode(nd.l), r;
          if (nd.op === "&&") return (truthy(l) && truthy(evalNode(nd.r))) ? 1 : 0;
          if (nd.op === "||") return (truthy(l) || truthy(evalNode(nd.r))) ? 1 : 0;
          r = evalNode(nd.r);
          switch (nd.op) {
            case "xor": return (truthy(l) !== truthy(r)) ? 1 : 0;
            case "+": return l + r; case "-": return l - r; case "*": return l * r;
            case "/": return r === 0 ? 0 : l / r; case "%": return r === 0 ? 0 : ((l % r) + r) % r;
            case "=": return eqv(l, r) ? 1 : 0; case "<>": return eqv(l, r) ? 0 : 1;
            case "<": return l < r ? 1 : 0; case "<=": return l <= r ? 1 : 0;
            case ">": return l > r ? 1 : 0; case ">=": return l >= r ? 1 : 0;
          }
        }
      }
      throw new Error("cannot eval " + nd.type);
    }
    function evalRef(nd) { return nd; }
    function truthy(v) { return v !== 0 && v !== false && v !== "" && v != null; }
    function eqv(a, b) { if (typeof a === "string" || typeof b === "string") return String(a) === String(b); return a === b; }
    function callFn(name, a) {
      switch (name) {
        case "MIN": return Math.min.apply(null, a); case "MAX": return Math.max.apply(null, a);
        case "ABS": return Math.abs(a[0]); case "SQRT": return Math.sqrt(a[0]);
        case "REAL_TO_DINT": case "REAL_TO_INT": case "TRUNC": return Math.round(a[0]);
        case "LIMIT": return Math.max(a[0], Math.min(a[1], a[2]));
        case "SEL": return truthy(a[0]) ? a[2] : a[1];
        case "MUX": { var k = (a[0] | 0) + 1; if (k < 1) k = 1; if (k > a.length - 1) k = a.length - 1; return a[k]; }
        case "SHL": return (a[0] | 0) << (a[1] | 0);
        case "SHR": return (a[0] | 0) >> (a[1] | 0);
        case "MOVE": return a[0];
        case "LEN": return String(a[0]).length;
        case "CONCAT": return a.map(String).join("");
        case "LEFT": return String(a[0]).slice(0, a[1] | 0);
        case "RIGHT": { var nR = a[1] | 0; return nR <= 0 ? "" : String(a[0]).slice(-nR); }
        case "MID": return String(a[0]).substr((a[2] | 0) - 1, a[1] | 0);
        case "FIND": { var fi = String(a[0]).indexOf(String(a[1])); return fi < 0 ? 0 : fi + 1; }
        case "DELETE": { var sD = String(a[0]), lD = a[1] | 0, pD = a[2] | 0; return sD.slice(0, pD - 1) + sD.slice(pD - 1 + lD); }
        case "INSERT": { var sI = String(a[0]), s2 = String(a[1]), pI = a[2] | 0; return sI.slice(0, pI) + s2 + sI.slice(pI); }
        case "REPLACE": { var sR = String(a[0]), s2r = String(a[1]), lR = a[2] | 0, pR = a[3] | 0; return sR.slice(0, pR - 1) + s2r + sR.slice(pR - 1 + lR); }
        default:
          if (/_TO_/.test(name)) return a[0]; // DINT_TO_REAL, INT_TO_REAL, etc: passthrough
          return a[0];
      }
    }
    function assign(target, val) {
      if (target.type === "var") { env[K(target.name)] = val; return; }
      if (target.type === "field") { env[curPrefix + keyOf(target)] = val; return; }
      if (target.type === "index") {
        var bk = keyOf(target.base); var pbk = curPrefix + bk;
        var arr = evalNode(target.base); var i = evalNode(target.index) | 0;
        if (arr === 0 || arr === undefined) { arr = []; if (bk) env[pbk] = arr; }
        if (!Array.isArray(arr)) throw new Error("not an array: " + bk);
        if (i < 0) throw new Error("array index out of range [" + i + "]");
        if (i > 100000) throw new Error("array index too large [" + i + "]");
        while (arr.length <= i) arr.push(0);
        arr[i] = val; return;
      }
      throw new Error("bad assignment target");
    }
    function fbScanMs() { return (typeof env.dt === "number") ? env.dt : 100; }
    function runFB(inst, named) {
      var base = curPrefix + inst;
      var fb = env[base + ".__fb"]; if (!fb) { var uf = env[base + ".__ufb"]; if (uf && FBDEFS[uf]) runUserFB(base, FBDEFS[uf], named); return; }
      var dt = fbScanMs();
      function ip(k, dflt) { return named[k] !== undefined ? evalNode(named[k]) : dflt; }
      if (fb === "TON" || fb === "TOF" || fb === "TP") {
        var IN = truthy(ip("IN", env[base + ".IN"] || 0));
        var PT = Number(ip("PT", env[base + ".PT"] || 0));
        var ET = Number(env[base + ".ET"] || 0), Q = truthy(env[base + ".Q"] || 0);
        var prev = truthy(env[base + ".__prevIN"] || 0);
        env[base + ".IN"] = IN ? 1 : 0; env[base + ".PT"] = PT;
        if (fb === "TON") { if (IN) { ET = Math.min(PT, ET + dt); Q = ET >= PT; } else { ET = 0; Q = false; } }
        else if (fb === "TOF") { if (IN) { ET = 0; Q = true; } else if (Q) { ET = Math.min(PT, ET + dt); if (ET >= PT) Q = false; } }
        else { var timing = truthy(env[base + ".__timing"] || 0);
          if (IN && !prev && !timing) { timing = true; ET = 0; }
          if (timing) { ET = Math.min(PT, ET + dt); Q = true; if (ET >= PT) { Q = false; timing = false; } } else Q = false;
          if (!IN && !timing) ET = 0;
          env[base + ".__timing"] = timing ? 1 : 0;
        }
        env[base + ".ET"] = ET; env[base + ".Q"] = Q ? 1 : 0; env[base + ".__prevIN"] = IN ? 1 : 0;
      } else if (fb === "CTU") {
        var CU = truthy(ip("CU", 0)), R = truthy(ip("R", 0)), PV = Number(ip("PV", env[base + ".PV"] || 0));
        var CV = Number(env[base + ".CV"] || 0), pCU = truthy(env[base + ".__pCU"] || 0);
        if (R) CV = 0; else if (CU && !pCU) CV += 1;
        env[base + ".CV"] = CV; env[base + ".PV"] = PV; env[base + ".Q"] = CV >= PV ? 1 : 0; env[base + ".__pCU"] = CU ? 1 : 0;
      } else if (fb === "CTD") {
        var CD = truthy(ip("CD", 0)), LD = truthy(ip("LD", 0)), PVd = Number(ip("PV", env[base + ".PV"] || 0));
        var CVd = Number(env[base + ".CV"] || 0), pCD = truthy(env[base + ".__pCD"] || 0);
        if (LD) CVd = PVd; else if (CD && !pCD) CVd -= 1;
        env[base + ".CV"] = CVd; env[base + ".PV"] = PVd; env[base + ".Q"] = CVd <= 0 ? 1 : 0; env[base + ".__pCD"] = CD ? 1 : 0;
      } else if (fb === "R_TRIG") {
        var clkR = truthy(ip("CLK", 0)), mR = truthy(env[base + ".__m"] || 0);
        env[base + ".Q"] = (clkR && !mR) ? 1 : 0; env[base + ".__m"] = clkR ? 1 : 0;
      } else if (fb === "F_TRIG") {
        var clkF = truthy(ip("CLK", 0)), mF = truthy(env[base + ".__m"] || 0);
        env[base + ".Q"] = (!clkF && !mF) ? 1 : 0; env[base + ".__m"] = clkF ? 0 : 1;
      } else if (fb === "RS") {
        var setR = truthy(ip("S", 0)), rstR = truthy(ip("R1", 0)), qRS = truthy(env[base + ".Q"] || 0);
        qRS = (setR || qRS) && !rstR;
        env[base + ".Q"] = qRS ? 1 : 0; env[base + ".Q1"] = qRS ? 1 : 0;
      } else if (fb === "SR") {
        var setS = truthy(ip("S1", 0)), rstS = truthy(ip("R", 0)), qSR = truthy(env[base + ".Q"] || 0);
        qSR = setS || (qSR && !rstS);
        env[base + ".Q"] = qSR ? 1 : 0; env[base + ".Q1"] = qSR ? 1 : 0;
      } else if (fb === "CTUD") {
        var cu = truthy(ip("CU", 0)), cd = truthy(ip("CD", 0)), rst = truthy(ip("R", 0)), ld = truthy(ip("LD", 0)), pv = Number(ip("PV", env[base + ".PV"] || 0));
        var cv = Number(env[base + ".CV"] || 0), pcu = truthy(env[base + ".__pCU"] || 0), pcd = truthy(env[base + ".__pCD"] || 0);
        if (rst) cv = 0; else if (ld) cv = pv; else { if (cu && !pcu) cv += 1; if (cd && !pcd) cv -= 1; }
        env[base + ".CV"] = cv; env[base + ".PV"] = pv;
        env[base + ".QU"] = cv >= pv ? 1 : 0; env[base + ".QD"] = cv <= 0 ? 1 : 0;
        env[base + ".__pCU"] = cu ? 1 : 0; env[base + ".__pCD"] = cd ? 1 : 0;
      }
    }
    function evalInScope(node, base) { var sp = curPrefix; curPrefix = base + "."; try { return evalNode(node); } finally { curPrefix = sp; } }
    function initUserFB(base, def) {
      function ini(list) { for (var i = 0; i < list.length; i++) { var v = list[i], key = base + "." + v.name;
        if (v.fbType) { if (env[key + ".__fb"] === undefined) { env[key + ".__fb"] = v.fbType; env[key + ".Q"] = 0; env[key + ".ET"] = 0; env[key + ".CV"] = 0; } }
        else if (v.typeName && FBDEFS[v.typeName]) { if (env[key + ".__ufb"] === undefined) { env[key + ".__ufb"] = v.typeName; initUserFB(key, FBDEFS[v.typeName]); } }
        else if (env[key] === undefined) env[key] = (v.init !== undefined) ? evalInScope(v.init, base) : 0;
      } }
      ini(def.inputs); ini(def.outputs); ini(def.locals);
    }
    function runUserFn(def, args) {
      var pfx = "__fn" + (fnSeq++) + ".";
      for (var i = 0; i < def.inputs.length; i++) env[pfx + def.inputs[i].name] = args[i] === undefined ? 0 : args[i];
      for (var j = 0; j < def.locals.length; j++) { var lv = def.locals[j]; env[pfx + lv.name] = (lv.init !== undefined) ? evalInScope(lv.init, pfx.slice(0, -1)) : 0; }
      env[pfx + def.name] = 0;
      var sp = curPrefix; curPrefix = pfx;
      try { exec(def.body); } finally { curPrefix = sp; }
      var rv = env[pfx + def.name];
      for (var k in env) { if (k.indexOf(pfx) === 0) delete env[k]; }
      return rv === undefined ? 0 : rv;
    }
    function runUserFB(base, def, named) {
      if (env[base + ".__uinit"] === undefined) { initUserFB(base, def); env[base + ".__uinit"] = 1; }
      for (var i = 0; i < def.inputs.length; i++) { var inp = def.inputs[i], up = inp.name.toUpperCase(); if (named[up] !== undefined) env[base + "." + inp.name] = evalNode(named[up]); }
      var sp = curPrefix; curPrefix = base + ".";
      try { exec(def.body); } finally { curPrefix = sp; }
    }
    var BRK = {};
    function exec(list) {
      for (var k = 0; k < list.length; k++) {
        var s = list[k]; tick();
        switch (s.type) {
          case "decls": s.decls.forEach(function (d) {
            var dk = K(d.name);
            if (d.fbType) { if (env[dk + ".__fb"] === undefined) { env[dk + ".__fb"] = d.fbType; env[dk + ".Q"] = 0; env[dk + ".ET"] = 0; env[dk + ".CV"] = 0; } }
            else if (d.typeName && FBDEFS[d.typeName]) { if (env[dk + ".__ufb"] === undefined) { env[dk + ".__ufb"] = d.typeName; initUserFB(dk, FBDEFS[d.typeName]); } }
            else if (d.init !== undefined && env[dk] === undefined) env[dk] = evalNode(d.init);
          }); break;
          case "fbdef": break;
          case "fndef": break;
          case "assign": assign(s.target, evalNode(s.value)); break;
          case "exprstmt": if (s.expr && s.expr.type === "call" && s.expr.named) runFB(s.expr.name, s.expr.named); else evalNode(s.expr); break;
          case "if": {
            var done = false;
            for (var ci = 0; ci < s.clauses.length; ci++) { if (truthy(evalNode(s.clauses[ci].cond))) { exec(s.clauses[ci].body); done = true; break; } }
            if (!done && s.els) exec(s.els);
            break;
          }
          case "case": {
            var sv = evalNode(s.sel), hit = false;
            for (var bi = 0; bi < s.branches.length; bi++) { if (s.branches[bi].labels.some(function (L) { return eqv(L, sv); })) { exec(s.branches[bi].body); hit = true; break; } }
            if (!hit && s.els) exec(s.els);
            break;
          }
          case "for": {
            var from = evalNode(s.from) | 0, to = evalNode(s.to) | 0, by = s.by ? evalNode(s.by) : 1;
            if (by === 0) throw new Error("FOR step is 0");
            if (by > 0) { for (var iv = from; iv <= to; iv += by) { env[K(s.v)] = iv; try { exec(s.body); } catch (e) { if (e === BRK) break; throw e; } tick(); } }
            else { for (var iv2 = from; iv2 >= to; iv2 += by) { env[K(s.v)] = iv2; try { exec(s.body); } catch (e2) { if (e2 === BRK) break; throw e2; } tick(); } }
            break;
          }
          case "while": { while (truthy(evalNode(s.cond))) { try { exec(s.body); } catch (e3) { if (e3 === BRK) break; throw e3; } tick(); } break; }
          case "repeat": { do { try { exec(s.body); } catch (e4) { if (e4 === BRK) break; throw e4; } tick(); } while (!truthy(evalNode(s.cond))); break; }
          case "exit": throw BRK;
          case "return": return;
        }
      }
    }
    exec(prog);
    return env;
  }

  // Public: interpret ST source with a set of cases.
  // case = { set:{...}, scans:N, expect:{...}, name:"" }
  //   or with input transitions: { set:{...init}, steps:[{set:{...},scans:N}, ...], expect:{...} }
  // returns { results:[{name,pass,expected,actual,error}], fatal? }
  window.runSt = function (src, cases) {
    var prog;
    try { prog = parse(tokenize(src)); }
    catch (e) { return { fatal: String(e.message || e) }; }
    var results = [];
    for (var i = 0; i < cases.length; i++) {
      var tc = cases[i], env = {};
      // seed inputs (deep-copy arrays so cases don't share state)
      var set = tc.set || {};
      for (var kk in set) { env[kk] = Array.isArray(set[kk]) ? set[kk].slice() : set[kk]; }
      var r = { name: tc.name || ("case " + (i + 1)), pass: false };
      try {
        if (tc.steps && tc.steps.length) {
          for (var si = 0; si < tc.steps.length; si++) {
            var stp = tc.steps[si], sset = stp.set || {};
            for (var skk in sset) { env[skk] = Array.isArray(sset[skk]) ? sset[skk].slice() : sset[skk]; }
            var ssc = stp.scans || 1;
            for (var q = 0; q < ssc; q++) run(prog, env);
          }
        } else {
          var scans = tc.scans || 1;
          for (var sc = 0; sc < scans; sc++) run(prog, env);
        }
        var exp = tc.expect || {}, ok = true, ea = [], aa = [];
        for (var ek in exp) {
          var want = exp[ek], got = env[ek];
          ea.push(ek + "=" + fmtV(want)); aa.push(ek + "=" + fmtV(got));
          if (!approxEq(want, got)) ok = false;
        }
        r.pass = ok; r.expected = ea.join(", "); r.actual = aa.join(", ");
      } catch (err) { r.error = String(err.message || err); }
      results.push(r);
    }
    return { results: results };
  };

  function approxEq(a, b) {
    if (Array.isArray(a)) { if (!Array.isArray(b) || a.length !== b.length) return false; for (var i = 0; i < a.length; i++) if (!approxEq(a[i], b[i])) return false; return true; }
    if (typeof a === "boolean") a = a ? 1 : 0;
    if (typeof b === "boolean") b = b ? 1 : 0;
    if (typeof a === "number" && typeof b === "number") { if (Number.isInteger(a) && Number.isInteger(b)) return a === b; return Math.abs(a - b) < 1e-3; }
    return String(a) === String(b);
  }
  function fmtV(v) { if (typeof v === "boolean") return v ? "TRUE" : "FALSE"; if (typeof v === "number" && !Number.isInteger(v)) return v.toFixed(3); return String(v); }
})();

      h+='<details class="llreveal"><summary>Reveal reference solution</summary><pre class="llsol">'+_llEsc(l.solution)+'</pre>'+(l.explain?'<div class="llexpl">'+l.explain+'</div>':'')+'</details>';
    } else if(g==='diag'){
      h+='<label class="lllbl">Your diagnosis:</label>';
      h+='<div class="llrow"><input id="llin-'+l.id+'" class="llinput" placeholder="type the fault / fix..."><button class="btn" onclick="llCheckPredict(\u0027'+l.id+'\u0027)">Check</button></div>';
      h+='<div id="llout-'+l.id+'" class="llout"></div>';
      h+='<details class="llreveal"><summary>Reveal answer</summary><div class="llexpl"><b>'+_llEsc(l.expectedOutput)+'</b></div>'+(l.explain?'<div class="llexpl">'+l.explain+'</div>':'')+'</details>';
    } else {
      if(l.starter) h+='<label class="lllbl">Scratch pad:</label><textarea id="llta-'+l.id+'" class="llcode" spellcheck="false">'+_llEsc(l.starter)+'</textarea>';
      h+='<details class="llreveal"><summary>Reveal solution</summary><pre class="llsol">'+_llEsc(l.solution)+'</pre>'+(l.explain?'<div class="llexpl">'+l.explain+'</div>':'')+'</details>';
    }
    h+='</div></details>';
  }
  if(shown===0) h+='<p class="sub">No challenges in this filter.</p>';
  host.innerHTML=h;
}
